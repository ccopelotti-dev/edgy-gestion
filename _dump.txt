import { Vencimientos } from '@/pages/Vencimientos'

function App() {
  return (
    <TreasuryProvider>

            <Route path="/vencimientos" element={<Vencimientos />} />
          </Route>
        </Routes>
      </BrowserRouter>
    </TreasuryProvider>
  )
}

export default App
